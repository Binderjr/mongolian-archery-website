  <script>
    const shopItems = [
      { name: "Арьсан дээл", price: 280000, img: "https://via.placeholder.com/260x200?text=Leather+Deel" },
      { name: "Хантааз", price: 180000, img: "https://via.placeholder.com/260x200?text=Vest" },
      { name: "Морин гутал", price: 320000, img: "https://via.placeholder.com/260x200?text=Boots" },
      { name: "Нум, сумаар иж бүрдэл", price: 450000, img: "https://via.placeholder.com/260x200?text=Archery+Set" },
    ];

    const cart = [];

    const fmt = n => n.toLocaleString('mn-MN') + ' ₮';

    const renderShop = () => {
      const shop = document.getElementById('shopItems');
      shop.innerHTML = '';
      shopItems.forEach((item, idx) => {
        shop.innerHTML += `
          <div class="item">
            <img src="${item.img}" alt="${item.name}">
            <h3>${item.name}</h3>
            <p class="price">${fmt(item.price)}</p>
            <button class="add-btn" onclick="addToCart(${idx})">Сагсанд нэмэх</button>
          </div>
        `;
      });
    };

    const renderCart = () => {
      const cartDiv = document.getElementById('cart');
      cartDiv.innerHTML = cart.map((item, i) =>
        `<p>${item.name} — ${fmt(item.price)}</p>`
      ).join('') || "<p>Сагс хоосон байна</p>";
      computeTotals();
    };

    const addToCart = (idx) => {
      cart.push(shopItems[idx]);
      renderCart();
    };

    const computeTotals = () => {
      const subtotal = cart.reduce((a, b) => a + b.price, 0);
      const shippingMethod = document.getElementById('shippingMethod').value;
      let shipping = shippingMethod === 'express' ? 20000 : (subtotal >= 150000 ? 0 : 7000);
      const tax = subtotal * 0.1;
      const total = subtotal + shipping + tax;

      document.getElementById('totals').innerHTML = `
        <p>Нийт: ${fmt(subtotal)}</p>
        <p>НӨАТ (10%): ${fmt(tax)}</p>
        <p>Хүргэлт: ${fmt(shipping)}</p>
        <h3>Төлөх дүн: ${fmt(total)}</h3>
      `;
    };

    document.getElementById('shippingMethod').addEventListener('change', computeTotals);
    document.getElementById('checkoutBtn').addEventListener('click', () => {
      alert('Захиалга амжилттай илгээгдлээ! Та холбогдох имэйлээр хариу хүлээнэ үү.');
    });

    renderShop();
    renderCart();
  </script>
